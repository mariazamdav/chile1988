# Creating the county level variables 

# Proportion of bureaucrats per county
prop_burocrata <- Chilenos_Censo_1987 %>%
  group_by(Comuna_1987_Origen3) %>%
  summarise(prop_burocrata = as.numeric(mean(burocrata, na.rm = TRUE)))

# Proportion of bureaucrats per rango per comuna
prop_rango <- Chilenos_Censo_1987 %>%
  group_by(Comuna_1987_Origen3) %>%
  summarise(
    prop_high = as.numeric(mean(burocrata & rango == "high", na.rm = TRUE)),
    prop_mid  = as.numeric(mean(burocrata & rango == "mid",  na.rm = TRUE)),
    prop_low  = as.numeric(mean(burocrata & rango == "low",  na.rm = TRUE))
  )

# Getting population data from closer to the plebiscite 
pop_1987 <- Chilenos_Censo_1987 %>%
  group_by(Comuna_1987_Origen3) %>%
  summarise(pop_1987 = n())

# Joining the county level variables to the voting data databases 
VotingData_Dataverse <- VotingData_Dataverse %>%
  mutate(ID_2002_num = as.numeric(ID_2002)) %>%
  left_join(prop_burocrata, by = c("ID_2002_num" = "Comuna_1987_Origen3")) %>%
  left_join(prop_rango,     by = c("ID_2002_num" = "Comuna_1987_Origen3")) %>%
  left_join(pop_1987,       by = c("ID_2002_num" = "Comuna_1987_Origen3")) %>%
  select(-ID_2002_num) %>%
  mutate(
    sh_victims_total = (shvictimas_until74_70 + shvictimas_75_plus_70),
        prop_burocrata = prop_burocrata * 100,
        prop_high      = prop_high * 100,
        prop_mid       = prop_mid  * 100,
        prop_low       = prop_low  * 100) %>%
    mutate(share_high_among_buro = (prop_high / prop_burocrata) *100,
    share_mid_among_buro  = (prop_mid  / prop_burocrata) *100)

tidy_model <- function(model, model_name, keep_terms) {
  as.data.frame(summary(model)$coefficients) %>%
    tibble::rownames_to_column("term") %>%
    rename(estimate = Estimate, se = `Std. Error`, ci_low = `CI Lower`, 
    ci_high = `CI Upper`) %>%
    filter(term %in% keep_terms) %>%
    mutate(model = model_name)
}
